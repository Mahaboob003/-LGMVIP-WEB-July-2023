const taskList = document.getElementById('taskList'); const newTaskInput = document.getElementById('newTask'); function addTask() { const taskText = newTaskInput.value.trim(); if (taskText === '') return; const li = document.createElement('li'); li.innerText = taskText; const deleteButton = document.createElement('button'); deleteButton.innerText = 'Delete'; deleteButton.onclick = function() { li.remove(); }; li.appendChild(deleteButton); taskList.appendChild(li); newTaskInput.value = ''; } newTaskInput.addEventListener('keyup', function(event) { if (event.key === 'Enter') { addTask(); } });
